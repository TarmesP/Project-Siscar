CREATE DATABASE SISCARDB;
USE SISCARDB;

CREATE TABLE TBLMARCAR(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	MARID INT, 
	MARNOME VARCHAR(50),
	MAROBSERVACOES VARCHAR(500)
);

CREATE TABLE TBLVEICULOS(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	VEIID INT,
	VEIANOFABRICACAO INT,
	VEIANOMODELO INT,
	VEINOME VARCHAR(50),
	VEIOBSERVACOES VARCHAR(500)
);

CREATE TABLE TBLMODELOS(
	ID INT IDENTITY(1,1) PRIMARY KEY,
	MODID INT,
	MODNOME VARCHAR(50),
	MODOBSERVACOES VARCHAR(500)
);

CREATE TABLE TBLMODELOSVEICULOS(
	MODID INT,
	VEIID INT
);
